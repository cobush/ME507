395  void updateRPY()
396  {
397  a12 = 2.0f * (q[1] * q[2] + q[0] * q[3]);
398  a22 = q[0] * q[0] + q[1] * q[1] - q[2] * q[2] - q[3] * q[3];
399  a31 = 2.0f * (q[0] * q[1] + q[2] * q[3]);
400  a32 = 2.0f * (q[1] * q[3] - q[0] * q[2]);
401  a33 = q[0] * q[0] - q[1] * q[1] - q[2] * q[2] + q[3] * q[3];
402  pitch = -asinf(a32);
403  roll = atan2f(a31, a33);
404  yaw = atan2f(a12, a22);
405  pitch *= 180.0f / PI;
406  roll *= 180.0f / PI;
407  yaw *= 180.0f / PI;
408  yaw += magnetic_declination;
409  if (yaw >= +180.f) yaw -= 360.f;
410  else if (yaw < -180.f) yaw += 360.f;
411 
412  lin_ax = a[0] + a31;
413  lin_ay = a[1] + a32;
414  lin_az = a[2] - a33;
415  }

522  void initMPU9250()
523  {
524  // wake up device
525  writeByte(MPU9250_ADDRESS, PWR_MGMT_1, 0x00); // Clear sleep mode bit (6), enable all sensors
526  delay(100); // Wait for all registers to reset
527 
528  // get stable time source
529  writeByte(MPU9250_ADDRESS, PWR_MGMT_1, 0x01); // Auto select clock source to be PLL gyroscope reference if ready else
530  delay(200);
531 
532  // Configure Gyro and Thermometer
533  // Disable FSYNC and set thermometer and gyro bandwidth to 41 and 42 Hz, respectively;
534  // minimum delay time for this setting is 5.9 ms, which means sensor fusion update rates cannot
535  // be higher than 1 / 0.0059 = 170 Hz
536  // DLPF_CFG = bits 2:0 = 011; this limits the sample rate to 1000 Hz for both
537  // With the MPU9250, it is possible to get gyro sample rates of 32 kHz (!), 8 kHz, or 1 kHz
538  writeByte(MPU9250_ADDRESS, MPU_CONFIG, 0x03);
539 
540  // Set sample rate = gyroscope output rate/(1 + SMPLRT_DIV)
541  writeByte(MPU9250_ADDRESS, SMPLRT_DIV, 0x04); // Use a 200 Hz rate; a rate consistent with the filter update rate
542  // determined inset in CONFIG above
543 
544  // Set gyroscope full scale range
545  // Range selects FS_SEL and GFS_SEL are 0 - 3, so 2-bit values are left-shifted into positions 4:3
546  uint8_t c = readByte(MPU9250_ADDRESS, GYRO_CONFIG); // get current GYRO_CONFIG register value
547  // c = c & ~0xE0; // Clear self-test bits [7:5]
548  c = c & ~0x03; // Clear Fchoice bits [1:0]
549  c = c & ~0x18; // Clear GFS bits [4:3]
550  c = c | (uint8_t)GFSSEL << 3; // Set full scale range for the gyro
551  // c =| 0x00; // Set Fchoice for the gyro to 11 by writing its inverse to bits 1:0 of GYRO_CONFIG
552  writeByte(MPU9250_ADDRESS, GYRO_CONFIG, c ); // Write new GYRO_CONFIG value to register
553 
554  // Set accelerometer full-scale range configuration
555  c = readByte(MPU9250_ADDRESS, ACCEL_CONFIG); // get current ACCEL_CONFIG register value
556  // c = c & ~0xE0; // Clear self-test bits [7:5]
557  c = c & ~0x18; // Clear AFS bits [4:3]
558  c = c | (uint8_t)AFSSEL << 3; // Set full scale range for the accelerometer
559  writeByte(MPU9250_ADDRESS, ACCEL_CONFIG, c); // Write new ACCEL_CONFIG register value
560 
561  // Set accelerometer sample rate configuration
562  // It is possible to get a 4 kHz sample rate from the accelerometer by choosing 1 for
563  // accel_fchoice_b bit [3]; in this case the bandwidth is 1.13 kHz
564  c = readByte(MPU9250_ADDRESS, ACCEL_CONFIG2); // get current ACCEL_CONFIG2 register value
565  c = c & ~0x0F; // Clear accel_fchoice_b (bit 3) and A_DLPFG (bits [2:0])
566  c = c | 0x03; // Set accelerometer rate to 1 kHz and bandwidth to 41 Hz
567  writeByte(MPU9250_ADDRESS, ACCEL_CONFIG2, c); // Write new ACCEL_CONFIG2 register value
568 
569  // The accelerometer, gyro, and thermometer are set to 1 kHz sample rates,
570  // but all these rates are further reduced by a factor of 5 to 200 Hz because of the SMPLRT_DIV setting
571 
572  // Configure Interrupts and Bypass Enable
573  // Set interrupt pin active high, push-pull, hold interrupt pin level HIGH until interrupt cleared,
574  // clear on read of INT_STATUS, and enable I2C_BYPASS_EN so additional chips
575  // can join the I2C bus and all can be controlled by the Arduino as master
576  writeByte(MPU9250_ADDRESS, INT_PIN_CFG, 0x22);
577  writeByte(MPU9250_ADDRESS, INT_ENABLE, 0x01); // Enable data ready (bit 0) interrupt
578  delay(100);
579  }
